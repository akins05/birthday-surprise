  <script>
    "use strict";

    // --------- Customize your message here ----------
    const CONFIG = {
  from: "From: Your Daughter (OLUWAGBEMISOLA)",
  message:
`Happy Birthday Mom,

Thank you for loving me, praying for me, and standing by me through everything.
You are the heart of our home, and your sacrifices mean more than I can ever explain.

Today I celebrate you, your strength, your kindness, your beautiful smile, and your love.
May God bless you with long life, good health, peace of mind, and joy that will never finish.

I love you more than words can say.
Happy Birthday, Mom.`,
  autoSlideMs: 3500
};

    // Add your celebrant photos here (filenames must match your folder)
    const PHOTOS = [
      "images/p1.jpg",
      "images/p2.jpg",
      "images/p3.jpg",
      "images/p4.jpg",
      "images/p5.jpg"
    ];
    // -----------------------------------------------

    // Personalize with ?to=Name
    const params = new URLSearchParams(location.search);
    const toName = params.get("to");

    const startCard = document.getElementById("startCard");
    const surpriseCard = document.getElementById("surpriseCard");
    const openBtn = document.getElementById("openBtn");
    const replayBtn = document.getElementById("replayBtn");
    const copyBtn = document.getElementById("copyBtn");
    const musicBtn = document.getElementById("musicBtn");

    const toLine = document.getElementById("toLine");
    const typeTarget = document.getElementById("typeTarget");
    const signature = document.getElementById("signature");

    const audio = document.getElementById("audio");
    let musicOn = false;

    // Title lines
    document.getElementById("startTitle").textContent =
      toName ? `Birthday Surprise for ${toName}` : "Birthday Surprise";
    toLine.textContent = toName ? `To: ${toName}` : "To someone special";
    signature.textContent = CONFIG.from;

    // Balloons
    const balloons = document.getElementById("balloons");
    const colors = ["#ff4d8d", "#ffd166", "#4dd8ff", "#a5ff7a", "#c77dff"];
    function spawnBalloons(count = 14){
      balloons.innerHTML = "";
      for (let i=0;i<count;i++){
        const b = document.createElement("div");
        b.className = "balloon";
        b.style.left = Math.random()*100 + "vw";
        b.style.background = colors[Math.floor(Math.random()*colors.length)];
        b.style.animationDuration = (10 + Math.random()*10) + "s";
        b.style.animationDelay = (-Math.random()*10) + "s";
        b.style.opacity = (0.65 + Math.random()*0.25).toFixed(2);
        b.style.width = (54 + Math.random()*30) + "px";
        b.style.height = (70 + Math.random()*40) + "px";
        balloons.appendChild(b);
      }
    }
    spawnBalloons();

    // Typewriter (safe)
    function typeText(el, text, speed = 16){
      el.textContent = "";
      let i = 0;
      return new Promise(resolve => {
        const t = setInterval(() => {
          el.textContent += text[i] ?? "";
          i++;
          if (i >= text.length){
            clearInterval(t);
            resolve();
          }
        }, speed);
      });
    }

    // Confetti
    const canvas = document.getElementById("confetti");
    const ctx = canvas.getContext("2d");
    let W, H, confetti = [], confettiAnim = null;

    function resize(){
      W = canvas.width = window.innerWidth;
      H = canvas.height = window.innerHeight;
    }
    window.addEventListener("resize", resize);
    resize();

    function burstConfetti(pieces = 220){
      confetti = [];
      for (let i=0;i<pieces;i++){
        confetti.push({
          x: W/2 + (Math.random()-0.5)*120,
          y: H*0.25 + (Math.random()-0.5)*60,
          vx: (Math.random()-0.5)*8,
          vy: (Math.random()*-6) - 3,
          g: 0.16 + Math.random()*0.18,
          s: 4 + Math.random()*6,
          r: Math.random()*Math.PI,
          vr: (Math.random()-0.5)*0.2,
          c: colors[Math.floor(Math.random()*colors.length)],
          life: 260 + Math.random()*120
        });
      }
      if (confettiAnim) cancelAnimationFrame(confettiAnim);
      animateConfetti();
    }

    function animateConfetti(){
      ctx.clearRect(0,0,W,H);
      for (const p of confetti){
        p.life -= 1;
        p.vy += p.g;
        p.x += p.vx;
        p.y += p.vy;
        p.r += p.vr;

        ctx.save();
        ctx.translate(p.x, p.y);
        ctx.rotate(p.r);
        ctx.fillStyle = p.c;
        ctx.fillRect(-p.s/2, -p.s/2, p.s, p.s*1.4);
        ctx.restore();
      }
      confetti = confetti.filter(p => p.life > 0 && p.y < H + 60);
      confettiAnim = requestAnimationFrame(animateConfetti);
    }

    // Gallery slider (safe)
    const galleryImg = document.getElementById("galleryImg");
    const dotsWrap = document.getElementById("dots");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");

    let idx = 0;
    let slideTimer = null;

    function setImg(i){
      if (!PHOTOS.length) return;
      idx = (i + PHOTOS.length) % PHOTOS.length;

      galleryImg.src = PHOTOS[idx];
      // If an image path is wrong, show a readable fallback instead of breaking
      galleryImg.onerror = () => {
        galleryImg.onerror = null;
        galleryImg.alt = "Image not found. Check file name/path.";
        galleryImg.src =
          "data:image/svg+xml;charset=utf-8," +
          encodeURIComponent(`<svg xmlns="http://www.w3.org/2000/svg" width="800" height="1000">
            <rect width="100%" height="100%" fill="#111"/>
            <text x="50%" y="50%" fill="#fff" font-size="28" font-family="Arial" text-anchor="middle">
              Image not found
            </text>
            <text x="50%" y="56%" fill="#ccc" font-size="18" font-family="Arial" text-anchor="middle">
              Check images/p?.jpg
            </text>
          </svg>`);
      };

      // dots
      const dots = dotsWrap.querySelectorAll(".dot");
      dots.forEach((d, n) => d.classList.toggle("active", n === idx));
    }

    function buildDots(){
      dotsWrap.innerHTML = "";
      PHOTOS.forEach((_, n) => {
        const dot = document.createElement("div");
        dot.className = "dot" + (n === 0 ? " active" : "");
        dot.addEventListener("click", () => {
          stopAutoSlide();
          setImg(n);
          startAutoSlide();
        });
        dotsWrap.appendChild(dot);
      });
    }

    function startAutoSlide(){
      stopAutoSlide();
      if (PHOTOS.length <= 1) return;
      slideTimer = setInterval(() => setImg(idx + 1), CONFIG.autoSlideMs);
    }

    function stopAutoSlide(){
      if (slideTimer){
        clearInterval(slideTimer);
        slideTimer = null;
      }
    }

    prevBtn.addEventListener("click", () => { stopAutoSlide(); setImg(idx - 1); startAutoSlide(); });
    nextBtn.addEventListener("click", () => { stopAutoSlide(); setImg(idx + 1); startAutoSlide(); });

    // Preload images (reduces “blank” flashes)
    function preload(list){
      for (const src of list){
        const im = new Image();
        im.src = src;
      }
    }
    preload(PHOTOS);

    // Buttons
    async function openSurprise(){
      startCard.classList.add("hidden");
      surpriseCard.classList.remove("hidden");

      burstConfetti();

      // Init gallery
      buildDots();
      setImg(0);
      startAutoSlide();

      await typeText(typeTarget, CONFIG.message, 16);
    }

    openBtn.addEventListener("click", openSurprise);
    replayBtn.addEventListener("click", () => burstConfetti());

    copyBtn.addEventListener("click", async () => {
      try{
        await navigator.clipboard.writeText(location.href);
        copyBtn.textContent = "Link copied";
        setTimeout(()=>copyBtn.textContent="Copy this link", 1400);
      }catch(e){
        prompt("Copy this link:", location.href);
      }
    });

    musicBtn.addEventListener("click", async () => {
      musicOn = !musicOn;
      if (musicOn){
        try { await audio.play(); } catch(e) {}
        musicBtn.textContent = "Music: On";
      } else {
        audio.pause();
        musicBtn.textContent = "Music: Off";
      }
    });
  </script>
